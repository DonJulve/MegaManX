320struct Rect {
321 public:
322 float x; // x-coordinate of top-left corner
323 float y; // y-coordinate of top-left corner
324 float w; // width
325 float h; // height
326
327 // Default constructor
328 Rect() : x(0.0f), y(0.0f), w(0.0f), h(0.0f) {}
329
330 // Parameterized constructor
331 Rect(float xPos, float yPos, float width, float height)
332 : x(xPos), y(yPos), w(width), h(height) {}
333
334 // Method to calculate area
335 float getArea() const {
336 return w * h;
337 }
338
339 // Method to check if a point is inside the rectangle
340 bool contains(float pointX, float pointY) const {
341 return (pointX >= x && pointX <= x + w &&
342 pointY >= y && pointY <= y + h);
343 }
344
345 // Static method to check if two rectangles intersect
346 static bool hasIntersection(const Rect& rect1, const Rect& rect2) {
347 // Two rectangles don't intersect if one is to the left of the other
348 // or one is above the other
349 return !(rect1.x + rect1.w < rect2.x ||
350 rect2.x + rect2.w < rect1.x ||
351 rect1.y + rect1.h < rect2.y ||
352 rect2.y + rect2.h < rect1.y);
353 }
354
355 // Static method to get the intersection rectangle of two rectangles
356 static Rect intersectRect(const Rect& rect1, const Rect& rect2) {
357 // If there's no intersection, return a zero-sized rectangle
358 if (!hasIntersection(rect1, rect2)) {
359 return Rect();
360 }
361
362 // Calculate intersection coordinates
363 float left = std::max(rect1.x, rect2.x);
364 float right = std::min(rect1.x + rect1.w, rect2.x + rect2.w);
365 float top = std::max(rect1.y, rect2.y);
366 float bottom = std::min(rect1.y + rect1.h, rect2.y + rect2.h);
367
368 return Rect(left, top, right - left, bottom - top);
369 }
370};
